/** * Exercise 7: LocalStorage & Persistence * ====================================== * Managing state and persisting data in the browser. */ // --- Selectors --- const noteForm = document.querySelector('#note-form'); const titleInput = document.querySelector('#note-title'); const bodyInput = document.querySelector('#note-body'); const notesContainer = document.querySelector('#notes-container'); const notesCount = document.querySelector('#notes-count'); const searchInput = document.querySelector('#search-input'); const btnCancel = document.querySelector('#btn-cancel'); const formTitle = document.querySelector('#form-title'); // --- Application State --- let notes = JSON.parse(localStorage.getItem('myNotes')) || []; let editId = null; // Tracks if we are currently editing an existing note // ============================================================ // CORE FUNCTIONS // ============================================================ function saveToStorage() { localStorage.setItem('myNotes', JSON.stringify(notes)); renderNotes(); } function renderNotes(filter = '') { // Clear container notesContainer.innerHTML = ''; // Filter notes based on search input const filteredNotes = notes.filter(n => n.title.toLowerCase().includes(filter.toLowerCase()) || n.body.toLowerCase().includes(filter.toLowerCase()) ); notesCount.textContent = `${filteredNotes.length} notes found`; if (filteredNotes.length === 0) { notesContainer.innerHTML = '

No notes found. Start writing!

'; return; } // Create HTML for each note filteredNotes.forEach(note => { const noteCard = document.createElement('div'); noteCard.className = 'note-card'; noteCard.innerHTML = `

${note.title}

${note.body}

${new Date(note.date).toLocaleDateString()} `; notesContainer.appendChild(noteCard); }); } // ============================================================ // EVENT HANDLERS // ============================================================ noteForm.addEventListener('submit', (e) => { e.preventDefault(); const title = titleInput.value.trim(); const body = bodyInput.value.trim(); if (!title) return; if (editId) { // Update existing note notes = notes.map(n => n.id === editId ? { ...n, title, body, date: Date.now() } : n); editId = null; formTitle.textContent = "New Note"; btnCancel.classList.add('hidden'); } else { // Create new note const newNote = { id: Date.now().toString(), title, body, date: Date.now() }; notes.unshift(newNote); // Add to the beginning of the array } noteForm.reset(); saveToStorage(); }); function deleteNote(id) { if (confirm('Are you sure you want to delete this note?')) { notes = notes.filter(n => n.id !== id); saveToStorage(); } } function startEdit(id) { const noteToEdit = notes.find(n => n.id === id); if (!noteToEdit) return; // Fill form with current data titleInput.value = noteToEdit.title; bodyInput.value = noteToEdit.body; // Update UI state editId = id; formTitle.textContent = "Edit Note"; btnCancel.classList.remove('hidden'); titleInput.focus(); } btnCancel.addEventListener('click', () => { editId = null; noteForm.reset(); formTitle.textContent = "New Note"; btnCancel.classList.add('hidden'); }); searchInput.addEventListener('input', (e) => { renderNotes(e.target.value); }); // Initial Render renderNotes();
