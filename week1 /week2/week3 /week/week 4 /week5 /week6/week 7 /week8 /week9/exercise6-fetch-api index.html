/** * Exercise 6: Fetch & REST APIs * ============================= * Handling asynchronous requests with fetch() and async/await. */ // ============================================================ // TASK 1: Random Quote (GET Request) // API: https://api.quotable.io/random (or similar) // ============================================================ const quoteDisplay = document.querySelector('#quote-display'); const btnNewQuote = document.querySelector('#btn-new-quote'); async function fetchQuote() { quoteDisplay.innerHTML = '

Fetching inspiration...

'; try { // We use a fallback API in case one is down const response = await fetch('https://quotes-api-self.vercel.app/quote'); if (!response.ok) throw new Error('Network response was not ok'); const data = await response.json(); quoteDisplay.innerHTML = `
"${data.quote}"
— ${data.author} `; } catch (error) { quoteDisplay.innerHTML = `

Oops! ${error.message}

`; } } btnNewQuote.addEventListener('click', fetchQuote); // ============================================================ // TASK 2: GitHub User Search // ============================================================ const githubInput = document.querySelector('#github-input'); const btnSearch = document.querySelector('#btn-search-user'); const githubResult = document.querySelector('#github-result'); async function searchGithubUser() { const username = githubInput.value.trim(); if (!username) return; githubResult.innerHTML = '

Searching...

'; try { const response = await fetch(`https://api.github.com/users/${username}`); if (response.status === 404) throw new Error('User not found'); const user = await response.json(); githubResult.innerHTML = `

${user.name || user.login}

${user.bio || 'No bio available'}

View Profile
`; } catch (error) { githubResult.innerHTML = `

${error.message}

`; } } btnSearch.addEventListener('click', searchGithubUser); // ============================================================ // TASK 3: Posts Feed (JSONPlaceholder) // ============================================================ const postsContainer = document.querySelector('#posts-container'); const btnLoadMore = document.querySelector('#btn-load-more'); let postLimit = 3; async function loadPosts() { try { const response = await fetch(`https://jsonplaceholder.typicode.com/posts?_limit=${postLimit}`); const posts = await response.json(); postsContainer.innerHTML = posts.map(post => `

${post.title}

${post.body}

`).join(''); postLimit += 3; // Increase limit for next click } catch (error) { console.error('Error fetching posts:', error); } } btnLoadMore.addEventListener('click', loadPosts); loadPosts(); // Load initial posts // ============================================================ // TASK 5: Parallel Fetch (Promise.all) // ============================================================ const btnFetchAll = document.querySelector('#btn-fetch-all'); const multiResult = document.querySelector('#multi-result'); async function fetchAllData() { multiResult.innerHTML = '

Fetching multiple sources parallelly...

'; try { // Promise.all fires all requests at the same time! const [users, photos, todos] = await Promise.all([ fetch('https://jsonplaceholder.typicode.com/users/1').then(r => r.json()), fetch('https://jsonplaceholder.typicode.com/photos/1').then(r => r.json()), fetch('https://jsonplaceholder.typicode.com/todos/1').then(r => r.json()) ]); multiResult.innerHTML = `

User: ${users.name}

Photo Title: ${photos.title}

Latest Task: ${todos.title}

`; } catch (error) { multiResult.innerHTML = '

Failed to fetch all data.

'; } } btnFetchAll.addEventListener('click', fetchAllData);
